using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace inlUppg3
{
    /// <summary>
    /// TreeNodes that are used in the Binary Tree
    /// </summary>
    class TreeNode
    {
        public TreeNode left;
        public TreeNode right;
        public int value;
	public string msg;
	public bool used;
        public TreeNode(int value)
        {
            	this.value = value;
            	this.left = null;
            	this.right = null;
		this.msg = "";
		this.used = false;
        }
    }
}
